    Add On Click Event 

    [self.theButton A_Event_OnClick:^(id sender, id argument) {
            [((UIViewController*)argument).view
                setBackgroundColor:[UIColor A_ColorMakeFormString:@"ABABAB"]];
} WithObj:self];


    Remove On Click Event 

    [self.theButton A_Event_RemoveClick];


    Add an event 

    [self.theButton A_Event_Add:^(id sender, id argument) {
            NSLog(@"Test");
} forControlEvents:UIControlEventTouchDragEnter];


    Remove an event 

    [self.theButton A_Event_Remove:UIControlEventTouchDragEnter];
